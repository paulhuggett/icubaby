660 template <ICUBABY_CONCEPT_OUTPUT_ITERATOR (output_type) OutputIterator>
661 OutputIterator operator() (input_type code_unit, OutputIterator dest) noexcept {
662 if (code_unit < 0x80) {
663 *(dest++) = static_cast<output_type> (code_unit);
664 return dest;
665 }
666 if (code_unit < 0x800) {
667 return transcoder::write2 (code_unit, dest);
668 }
669 if (is_surrogate (code_unit)) {
670 return transcoder::not_well_formed (dest);
671 }
672 if (code_unit < 0x10000) {
673 return transcoder::write3 (code_unit, dest);
674 }
675 if (code_unit <= max_code_point) {
676 return transcoder::write4 (code_unit, dest);
677 }
678 return transcoder::not_well_formed (dest);
679 }

720 template <typename OutputIterator> static OutputIterator write2 (input_type code_unit, OutputIterator dest) {
721 *(dest++) = static_cast<output_type> ((code_unit >> 6U) | 0xc0U);
722 *(dest++) = static_cast<output_type> ((code_unit & 0x3fU) | 0x80U);
723 return dest;
724 }
730 template <typename OutputIterator> static OutputIterator write3 (input_type code_unit, OutputIterator dest) {
731 *(dest++) = static_cast<output_type> ((code_unit >> 12U) | 0xe0U);
732 *(dest++) = static_cast<output_type> (((code_unit >> 6U) & 0x3fU) | 0x80U);
733 *(dest++) = static_cast<output_type> ((code_unit & 0x3fU) | 0x80U);
734 return dest;
735 }
741 template <typename OutputIterator> static OutputIterator write4 (input_type code_unit, OutputIterator dest) {
742 *(dest++) = static_cast<output_type> ((code_unit >> 18U) | 0xf0U);
743 *(dest++) = static_cast<output_type> (((code_unit >> 12U) & 0x3fU) | 0x80U);
744 *(dest++) = static_cast<output_type> (((code_unit >> 6U) & 0x3fU) | 0x80U);
745 *(dest++) = static_cast<output_type> ((code_unit & 0x3fU) | 0x80U);

747 }

753 well_formed_ = false;

786 template <ICUBABY_CONCEPT_OUTPUT_ITERATOR (output_type) OutputIterator>
787 OutputIterator operator() (input_type code_unit, OutputIterator dest) {
788 // Prior to C++20, char8 might be signed.
789 static_assert (sizeof (input_type) == sizeof (std::uint8_t));
790 auto const ucu = static_cast<std::uint8_t> (code_unit);
791 static_assert (std::is_unsigned_v<decltype (ucu)> && std::numeric_limits<decltype (ucu)>::max () <= utf8d_.size ());
792 // NOLINTNEXTLINE(cppcoreguidelines-pro-bounds-constant-array-index)
793 auto const type = utf8d_[ucu];
794 code_point_ = (state_ != accept)
795 ? static_cast<std::uint_least32_t> (static_cast<std::byte> (code_unit) & std::byte{0x3FU}) |
796 static_cast<uint_least32_t> (code_point_ << 6U)
797 : (0xFFU >> type) & ucu;
798 auto const idx = 256U + state_ + type;
799 assert (idx < utf8d_.size ());
800 // NOLINTNEXTLINE(cppcoreguidelines-pro-bounds-constant-array-index)
801 state_ = utf8d_[idx];
802 switch (state_) {
803 case accept: *(dest++) = code_point_; break;
804 case reject:
805 well_formed_ = false;
806 state_ = accept;
807 *(dest++) = replacement_char;
808 break;
809 default: break;
810 }
811 return dest;
812 }

820 template <ICUBABY_CONCEPT_OUTPUT_ITERATOR (output_type) OutputIterator>
821 constexpr OutputIterator end_cp (OutputIterator dest) {
822 if (state_ != accept) {
823 state_ = reject;
824 *(dest++) = replacement_char;
825 well_formed_ = false;
826 }
827 return dest;
828 }

958template <> class transcoder<char16_t, char32_t> {
959public:
961 using input_type = char16_t;
963 using output_type = char32_t;
964
965 constexpr transcoder () noexcept : transcoder (true) {}
970 explicit constexpr transcoder (bool well_formed) noexcept
971 : high_{0},
972 has_high_{static_cast<uint_least16_t> (false)},
973 well_formed_{static_cast<uint_least16_t> (well_formed)} {}
974
982 template <ICUBABY_CONCEPT_OUTPUT_ITERATOR (output_type) OutputIterator>
983 OutputIterator operator() (input_type code_unit, OutputIterator dest) noexcept {
984 if (!has_high_) {
985 if (is_high_surrogate (code_unit)) {
986 // A high surrogate code unit indicates that this is the first of a
987 // high/low surrogate pair.
988 high_ = adjusted_high (code_unit);
989 has_high_ = true;
990 return dest;
991 }
992
993 // A low surrogate without a preceeding high surrogate.
994 if (is_low_surrogate (code_unit)) {
995 well_formed_ = false;
996 code_unit = replacement_char;
997 }
998 *(dest++) = code_unit;
999 return dest;
1000 }
1001
1002 // A high surrogate followed by a low surrogate.
1003 if (is_low_surrogate (code_unit)) {
1004 *(dest++) = (static_cast<char32_t> (high_) << high_bits) + (code_unit - first_low_surrogate) + 0x10000;
1005 high_ = 0;
1006 has_high_ = false;
1007 return dest;
1008 }
1009 // There was a high surrogate followed by something other than a low
1010 // surrogate. A high surrogate followed by a second high surrogate yields
1011 // a single REPLACEMENT CHARACTER. A high followed by something other than
1012 // a low surrogate gives REPLACEMENT CHARACTER followed by the second input
1013 // code point.
1014 *(dest++) = replacement_char;
1015 well_formed_ = false;
1016 if (is_high_surrogate (code_unit)) {
1017 // There was a high surrogate followed by a second high surrogate. Remember the later of the two.
1018 high_ = adjusted_high (code_unit);
1019 assert (has_high_);
1020 return dest;
1021 }
1022
1023 *(dest++) = code_unit;
1024 high_ = 0;
1025 has_high_ = false;
1026 return dest;
1027 }
1028
1034 template <ICUBABY_CONCEPT_OUTPUT_ITERATOR (output_type) OutputIterator> OutputIterator end_cp (OutputIterator dest) {
1035 if (has_high_) {
1036 *(dest++) = replacement_char;
1037 high_ = 0;
1038 has_high_ = false;
1039 well_formed_ = false;
1040 }
1041 return dest;
1042 }

1052 auto tcdr = dest.transcoder ();
1053 assert (tcdr == this);
1054 return {tcdr, tcdr->end_cp (dest.base ())};
1055 }

1058 [[nodiscard]] constexpr bool well_formed () const noexcept { return well_formed_; }
1061 [[nodiscard]] constexpr bool partial () const noexcept { return has_high_; }

1063private:
1065 static constexpr auto high_bits = 10U;
1068 uint_least16_t high_ : high_bits;
1071 uint_least16_t has_high_ : 1;
1074 uint_least16_t well_formed_ : 1;

1083 static std::uint_least16_t adjusted_high (std::uint_least16_t code_unit) noexcept {
1084 assert (code_unit >= first_high_surrogate && "A high surrogate must be at least first_high_surrogate");
1085 auto const high_cu = code_unit - first_high_surrogate;
1086 assert (high_cu < std::numeric_limits<decltype (high_)>::max () && high_cu < (1U << high_bits) &&
1087 "high_cu won't fit in the high_ field!");
