635 template <ICUBABY_CONCEPT_OUTPUT_ITERATOR(output_type) OutputIterator>
636 OutputIterator operator() (input_type code_unit, OutputIterator dest) noexcept {
637 if (code_unit < 0x80) {
638 *(dest++) = static_cast<output_type> (code_unit);
639 return dest;
640 }
641 if (code_unit < 0x800) {
642 return transcoder::write2 (code_unit, dest);
643 }
644 if (is_surrogate (code_unit)) {
645 return transcoder::not_well_formed (dest);
646 }
647 if (code_unit < 0x10000) {
648 return transcoder::write3 (code_unit, dest);
649 }
650 if (code_unit <= max_code_point) {
651 return transcoder::write4 (code_unit, dest);
652 }
653 return transcoder::not_well_formed (dest);
654 }

695 template <typename OutputIterator> static OutputIterator write2 (input_type code_unit, OutputIterator dest) {
696 *(dest++) = static_cast<output_type> ((code_unit >> 6U) | 0xc0U);
697 *(dest++) = static_cast<output_type> ((code_unit & 0x3fU) | 0x80U);

705 template <typename OutputIterator> static OutputIterator write3 (input_type code_unit, OutputIterator dest) {
706 *(dest++) = static_cast<output_type> ((code_unit >> 12U) | 0xe0U);
707 *(dest++) = static_cast<output_type> (((code_unit >> 6U) & 0x3fU) | 0x80U);
708 *(dest++) = static_cast<output_type> ((code_unit & 0x3fU) | 0x80U);
709 return dest;
710 }
716 template <typename OutputIterator> static OutputIterator write4 (input_type code_unit, OutputIterator dest) {
717 *(dest++) = static_cast<output_type> ((code_unit >> 18U) | 0xf0U);
718 *(dest++) = static_cast<output_type> (((code_unit >> 12U) & 0x3fU) | 0x80U);
719 *(dest++) = static_cast<output_type> (((code_unit >> 6U) & 0x3fU) | 0x80U);
720 *(dest++) = static_cast<output_type> ((code_unit & 0x3fU) | 0x80U);

722 }

728 well_formed_ = false;

761 template <ICUBABY_CONCEPT_OUTPUT_ITERATOR(output_type) OutputIterator>
762 OutputIterator operator() (input_type code_unit, OutputIterator dest) {
763 // Prior to C++20, char8 might be signed.
764 static_assert (sizeof (input_type) == sizeof (std::uint8_t));
765 auto const ucu = static_cast<std::uint8_t> (code_unit);
766 static_assert (std::is_unsigned_v<decltype (ucu)> && std::numeric_limits<decltype (ucu)>::max () <= utf8d_.size ());
767 // NOLINTNEXTLINE(cppcoreguidelines-pro-bounds-constant-array-index)
768 auto const type = utf8d_[ucu];
769 code_point_ = (state_ != accept)
770 ? static_cast<std::uint_least32_t> (static_cast<std::byte> (code_unit) & std::byte{0x3FU}) |
771 static_cast<uint_least32_t> (code_point_ << 6U)
772 : (0xFFU >> type) & ucu;
773 auto const idx = 256U + state_ + type;
774 assert (idx < utf8d_.size ());
775 // NOLINTNEXTLINE(cppcoreguidelines-pro-bounds-constant-array-index)
776 state_ = utf8d_[idx];
777 switch (state_) {
778 case accept: *(dest++) = code_point_; break;
779 case reject:
780 well_formed_ = false;
781 state_ = accept;
782 *(dest++) = replacement_char;
783 break;
784 default: break;
785 }
786 return dest;
787 }

798 state_ = reject;
799 *(dest++) = replacement_char;
800 well_formed_ = false;
801 }
802 return dest;

933template <> class transcoder<char16_t, char32_t> {
934public:
936 using input_type = char16_t;
938 using output_type = char32_t;
939
940 constexpr transcoder () noexcept : transcoder (true) {}
945 explicit constexpr transcoder (bool well_formed) noexcept
946 : high_{0},
947 has_high_{static_cast<uint_least16_t> (false)},
948 well_formed_{static_cast<uint_least16_t> (well_formed)} {}
949
957 template <ICUBABY_CONCEPT_OUTPUT_ITERATOR(output_type) OutputIterator>
958 OutputIterator operator() (input_type code_unit, OutputIterator dest) {
959 if (!has_high_) {
960 if (is_high_surrogate (code_unit)) {
961 // A high surrogate code unit indicates that this is the first of a
962 // high/low surrogate pair.
963 high_ = adjusted_high (code_unit);
964 has_high_ = true;
965 return dest;
966 }
967
968 // A low surrogate without a preceeding high surrogate.
969 if (is_low_surrogate (code_unit)) {
970 well_formed_ = false;
971 code_unit = replacement_char;
972 }
973 *(dest++) = code_unit;
974 return dest;
975 }
976
977 // A high surrogate followed by a low surrogate.
978 if (is_low_surrogate (code_unit)) {
979 *(dest++) = (static_cast<char32_t> (high_) << high_bits) + (code_unit - first_low_surrogate) + 0x10000;
980 high_ = 0;
981 has_high_ = false;
982 return dest;
983 }
984 // There was a high surrogate followed by something other than a low
985 // surrogate. A high surrogate followed by a second high surrogate yields
986 // a single REPLACEMENT CHARACTER. A high followed by something other than
987 // a low surrogate gives REPLACEMENT CHARACTER followed by the second input
988 // code point.
989 *(dest++) = replacement_char;
990 well_formed_ = false;
991 if (is_high_surrogate (code_unit)) {
992 // There was a high surrogate followed by a second high surrogate. Remember the later of the two.
993 high_ = adjusted_high (code_unit);
994 assert (has_high_);
995 return dest;
996 }
997
998 *(dest++) = code_unit;
999 high_ = 0;
1000 has_high_ = false;
1001 return dest;
1002 }
1003
1009 template <ICUBABY_CONCEPT_OUTPUT_ITERATOR(output_type) OutputIterator>
1010 OutputIterator end_cp (OutputIterator dest) {
1011 if (has_high_) {
1012 *(dest++) = replacement_char;
1013 high_ = 0;
1014 has_high_ = false;
1015 well_formed_ = false;
1016 }
1017 return dest;
1018 }

1029 assert (tcdr == this);
1030 return {tcdr, tcdr->end_cp (dest.base ())};
1031 }

1037 [[nodiscard]] constexpr bool partial () const noexcept { return has_high_; }

1059 static std::uint_least16_t adjusted_high (std::uint_least16_t code_unit) noexcept {
1060 assert (code_unit >= first_high_surrogate && "A high surrogate must be at least first_high_surrogate");
1061 auto const high_cu = code_unit - first_high_surrogate;
1062 assert (high_cu < std::numeric_limits<decltype (high_)>::max () && high_cu < (1U << high_bits) &&
1063 "high_cu won't fit in the high_ field!");
