630 template <ICUBABY_CONCEPT_OUTPUT_ITERATOR(output_type) OutputIterator>
631 OutputIterator operator() (input_type code_unit, OutputIterator dest) noexcept {
632 if (code_unit < 0x80) {
633 *(dest++) = static_cast<output_type> (code_unit);
634 return dest;
635 }
636 if (code_unit < 0x800) {
637 return transcoder::write2 (code_unit, dest);
638 }
639 if (is_surrogate (code_unit)) {
640 return transcoder::not_well_formed (dest);
641 }
642 if (code_unit < 0x10000) {
643 return transcoder::write3 (code_unit, dest);
644 }
645 if (code_unit <= max_code_point) {
646 return transcoder::write4 (code_unit, dest);
647 }
648 return transcoder::not_well_formed (dest);
649 }

684 template <typename OutputIterator> static OutputIterator write2 (input_type code_unit, OutputIterator dest) {
685 *(dest++) = static_cast<output_type> ((code_unit >> 6U) | 0xc0U);
686 *(dest++) = static_cast<output_type> ((code_unit & 0x3fU) | 0x80U);
687 return dest;
688 }
689 template <typename OutputIterator> static OutputIterator write3 (input_type code_unit, OutputIterator dest) {
690 *(dest++) = static_cast<output_type> ((code_unit >> 12U) | 0xe0U);
691 *(dest++) = static_cast<output_type> (((code_unit >> 6U) & 0x3fU) | 0x80U);
692 *(dest++) = static_cast<output_type> ((code_unit & 0x3fU) | 0x80U);
693 return dest;
694 }
695 template <typename OutputIterator> static OutputIterator write4 (input_type code_unit, OutputIterator dest) {
696 *(dest++) = static_cast<output_type> ((code_unit >> 18U) | 0xf0U);
697 *(dest++) = static_cast<output_type> (((code_unit >> 12U) & 0x3fU) | 0x80U);
698 *(dest++) = static_cast<output_type> (((code_unit >> 6U) & 0x3fU) | 0x80U);
699 *(dest++) = static_cast<output_type> ((code_unit & 0x3fU) | 0x80U);
700 return dest;
701 }

704 well_formed_ = false;
705 static_assert (!is_surrogate (replacement_char));

737 template <ICUBABY_CONCEPT_OUTPUT_ITERATOR(output_type) OutputIterator>
738 OutputIterator operator() (input_type code_unit, OutputIterator dest) {
739 // Prior to C++20, char8 might be signed.
740 static_assert (sizeof (input_type) == sizeof (std::uint8_t));
741 auto const ucu = static_cast<std::uint8_t> (code_unit);
742 static_assert (std::is_unsigned_v<decltype (ucu)> && std::numeric_limits<decltype (ucu)>::max () <= utf8d_.size ());
743 // NOLINTNEXTLINE(cppcoreguidelines-pro-bounds-constant-array-index)
744 auto const type = utf8d_[ucu];
745 code_point_ = (state_ != accept)
746 ? static_cast<std::uint_least32_t> (static_cast<std::byte> (code_unit) & std::byte{0x3FU}) |
747 static_cast<uint_least32_t> (code_point_ << 6U)
748 : (0xFFU >> type) & ucu;
749 auto const idx = 256U + state_ + type;
750 assert (idx < utf8d_.size ());
751 // NOLINTNEXTLINE(cppcoreguidelines-pro-bounds-constant-array-index)
752 state_ = utf8d_[idx];
753 switch (state_) {
754 case accept: *(dest++) = code_point_; break;
755 case reject:
756 well_formed_ = false;
757 state_ = accept;
758 *(dest++) = replacement_char;
759 break;
760 default: break;
761 }
762 return dest;
763 }

902template <> class transcoder<char16_t, char32_t> {
903public:
905 using input_type = char16_t;
907 using output_type = char32_t;
908
909 constexpr transcoder () noexcept : transcoder (true) {}
914 explicit constexpr transcoder (bool well_formed) noexcept
915 : high_{0},
916 has_high_{static_cast<uint_least16_t> (false)},
917 well_formed_{static_cast<uint_least16_t> (well_formed)} {}
918
926 template <ICUBABY_CONCEPT_OUTPUT_ITERATOR(output_type) OutputIterator>
927 OutputIterator operator() (input_type code_unit, OutputIterator dest) {
928 if (!has_high_) {
929 if (is_high_surrogate (code_unit)) {
930 // A high surrogate code unit indicates that this is the first of a
931 // high/low surrogate pair.
932 high_ = adjusted_high (code_unit);
933 has_high_ = true;
934 return dest;
935 }
936
937 // A low surrogate without a preceeding high surrogate.
938 if (is_low_surrogate (code_unit)) {
939 well_formed_ = false;
940 code_unit = replacement_char;
941 }
942 *(dest++) = code_unit;
943 return dest;
944 }
945
946 // A high surrogate followed by a low surrogate.
947 if (is_low_surrogate (code_unit)) {
948 *(dest++) = (static_cast<char32_t> (high_) << high_bits) + (code_unit - first_low_surrogate) + 0x10000;
949 high_ = 0;
950 has_high_ = false;
951 return dest;
952 }
953 // There was a high surrogate followed by something other than a low
954 // surrogate. A high surrogate followed by a second high surrogate yields
955 // a single REPLACEMENT CHARACTER. A high followed by something other than
956 // a low surrogate gives REPLACEMENT CHARACTER followed by the second input
957 // code point.
958 *(dest++) = replacement_char;
959 well_formed_ = false;
960 if (is_high_surrogate (code_unit)) {
961 // There was a high surrogate followed by a second high surrogate. Remember the later of the two.
962 high_ = adjusted_high (code_unit);
963 assert (has_high_);
964 return dest;
965 }
966
967 *(dest++) = code_unit;
968 high_ = 0;
969 has_high_ = false;
970 return dest;
971 }
972
978 template <ICUBABY_CONCEPT_OUTPUT_ITERATOR(output_type) OutputIterator>
979 OutputIterator end_cp (OutputIterator dest) {
980 if (has_high_) {
981 *(dest++) = replacement_char;
982 high_ = 0;
983 has_high_ = false;
984 well_formed_ = false;
985 }
986 return dest;
987 }

997 auto tcdr = dest.transcoder ();
998 assert (tcdr == this);
999 return {tcdr, tcdr->end_cp (dest.base ())};
1000 }

1003 [[nodiscard]] constexpr bool well_formed () const noexcept { return well_formed_; }
1006 [[nodiscard]] constexpr bool partial () const noexcept { return has_high_; }

1008private:
1009 static constexpr auto high_bits = 10U;
1012 uint_least16_t high_ : high_bits;
1015 uint_least16_t has_high_ : 1;
1018 uint_least16_t well_formed_ : 1;

1027 static std::uint_least16_t adjusted_high (std::uint_least16_t code_unit) noexcept {
1028 assert (code_unit >= first_high_surrogate && "A high surrogate must be at least first_high_surrogate");
1029 auto const high_cu = code_unit - first_high_surrogate;
1030 assert (high_cu < std::numeric_limits<decltype (high_)>::max () && high_cu < (1U << high_bits) &&
1031 "high_cu won't fit in the high_ field!");

1086 break;

1091 // bytes consumed so far.

1839 }
